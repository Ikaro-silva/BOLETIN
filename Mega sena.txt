from random import randint
lista = list()
jogos = list()

tot = 1
quant = int(input('NUMEROS DE JOGOS:'))
while tot <= quant:
    cont = 0
    while True:
        num = randint(1, 60)
        if num not in lista:
            lista.append(num)
            cont += 1
        if cont >= 6:
            break
    lista.sort()
    jogos. append(lista[:])
    lista.clear()
    tot +=1

for i, l  in enumerate(jogos):
    print(f'{i},{l}')
